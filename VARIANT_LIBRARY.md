## Usage Example

```typescript
import { getSectionComponent } from './sections/sectionRegistry';

// Get a specific variant
const HeroCinematic = getSectionComponent('hero', 'cinematic');
const WeddingPartyGrid = getSectionComponent('wedding-party', 'grid');
const DressCodeLuxury = getSectionComponent('dress-code', 'luxury');

// Use in template
const section = {
  type: 'wedding-party',
  variant: 'showcase',
  enabled: true,
  bindings: {},
  settings: {}
};

const Component = getSectionComponent(section.type, section.variant);
return <Component section={section} />;
```
